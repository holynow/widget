2) Script injection demo

This emulates adding a script tag to the theme footer. It only runs on product detail paths; here we force-run in dev.

Snippet template:

<script>
(function () {
  var isDetail = /product\/detail/.test(location.pathname);
  if (!isDetail) return;
  var s = document.createElement('script');
  s.type = 'module';
  s.defer = true;
  s.src = 'https://<your-pages-domain>/assets/index-<hash>.js';
  document.head.appendChild(s);
})();
</script>
